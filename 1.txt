  <script>
    const membersKey = 'oneearth_members_local_v1';
    const membersCount = document.getElementById('membersCount');
    const form = document.getElementById('memberForm');
    const formMsg = document.getElementById('formMsg');
    const year = document.getElementById('year');
    year.textContent = new Date().getFullYear();

    function loadMembers() {
      try { return JSON.parse(localStorage.getItem(membersKey) || '[]') } 
      catch { return [] }
    }
    function saveMembers(list) { localStorage.setItem(membersKey, JSON.stringify(list)) }
    function renderCount() { membersCount.textContent = loadMembers().length }

    form.addEventListener('submit', e => {
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const father = document.getElementById('father').value.trim();
      const address = document.getElementById('address').value.trim();
      const aadhaar = document.getElementById('aadhaar').value.trim();
      if(!name || !phone){ formMsg.textContent='নাম ও মোবাইল দিন'; return; }
      const members = loadMembers();
      const newM = { id: Date.now(), name, father, address, phone, aadhaar, joinedAt: new Date().toISOString() };
      members.unshift(newM);
      saveMembers(members);
      renderCount();
      formMsg.textContent='ধন্যবাদ — সদস্যপদ যুক্ত হয়েছে';
      form.reset();
      setTimeout(()=>formMsg.textContent='',2500);
    });

    document.getElementById('exportBtn').addEventListener('click', ()=>{
      const members = loadMembers();
      if(!members.length){ alert('কোনো সদস্য নেই'); return; }
      const header = ['ID','Name','Father','Address','Phone','Aadhaar','JoinedAt'];
      const rows = members.map(m => [m.id,m.name,m.father,m.address,m.phone,m.aadhaar,m.joinedAt]);
      const csv = [header,...rows].map(r=>r.map(c=>"${c||''}").join(',')).join('\n');
      const blob = new Blob([csv], {type:'text/csv'});
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url; a.download = 'oneearth_members.csv'; a.click();
      URL.revokeObjectURL(url);
    });

    document.getElementById('copyUpi').addEventListener('click', ()=>{
      const upi = document.getElementById('upiId').textContent;
      navigator.clipboard?.writeText(upi).then(()=> alert('UPI ID copied: '+upi)).catch(()=> alert('Copy failed, please copy manually'));
    });

    renderCount();
  </script>
